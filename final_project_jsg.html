# Pills bought by Pharmacies in Each county per Year #

pharma_df = left_join(prod_county, county_pop, by = c("buyer_county", "year")) %>% 
  select(county_name, year, count, population) %>% 
  rename(county = county_name,
         pills_bought = count) %>% 
  mutate(
    year = recode(year, 
                  "2006" = 0,
                  "2007" = 1,
                  "2008" = 2,
                  "2009" = 3,
                  "2010" = 4,
                  "2011" = 5,
                  "2012" = 6)
  ) %>% 
  group_by(county, year) %>% 
  summarize(ppp = pills_bought/population)
pharma_df %>% 
  ggplot(aes(x = year, y = ppp, color = county)) +
  geom_point() +
  geom_line(se = FALSE) +
  theme(legend.position = "none")

int_slope_df <- pharma_df %>% 
  nest(data = year:ppp) %>%
  mutate(
    models = map(data, ~lm(ppp ~ year, data = .x)),
    result = map(models, broom::tidy)
  ) %>% 
  select(county, result) %>% 
  unnest(result) %>% 
  select(county, term, estimate) %>% 
  pivot_wider(
    names_from = term,
    values_from = estimate
  ) %>% 
  rename(int = "(Intercept)", slope = year)
int_slope_df %>% 
  ggplot(aes(x = slope, y = int)) +
  geom_point() +
  theme(legend.position = "none")

km_fit = 
  kmeans(
    int_slope_df[,c(2,3)], 
    centers = 3)

int_slope_df =
  broom::augment(km_fit, int_slope_df)
int_slope_df %>% 
  ggplot(aes(x = int, y = slope, color = .cluster)) +
  geom_point()

left_join(pharma_df, int_slope_df) %>% 
  ggplot(aes(x = year, y = ppp, group = county, color = .cluster)) + 
  geom_point() + 
  geom_path()
